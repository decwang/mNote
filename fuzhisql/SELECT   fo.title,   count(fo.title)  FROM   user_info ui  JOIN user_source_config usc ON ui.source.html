SELECT
fo.title,
count(fo.title)
FROM
user_info ui
JOIN user_source_config usc ON ui.source = usc.id
JOIN fuma_info fi ON fi.user_id = ui.id
JOIN fuma_order fo ON fo.id = fi.order_id
WHERE
ui.create_time >= '2017-06-13 00:00:00'
AND ui.create_time <  '2017-06-19 00:00:00'
GROUP BY
fo.title
