#前置任务查询
SELECT A.ID,A.ACT_TITLE,,A.TTS_ACT_URL,A.TTS_ACT_IMAGEURL,A.TTS_ACT_TITLE,A.TASK_CONFIG_ID,B.NAME,B.CHANNEL_ID FROM TB_ACT_CONFIG A,TB_ACT_TASK_CONFIG B
WHERE 1=1
AND A.TASK_CONFIG_ID = B.ID
AND A.TASK_CONFIG_ID > 0
ORDER BY A.ID DESC;
