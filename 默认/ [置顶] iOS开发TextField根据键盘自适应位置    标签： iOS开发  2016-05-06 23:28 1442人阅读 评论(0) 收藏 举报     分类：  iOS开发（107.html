  1. - (void)setNotification {  
  2.     [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(keyboardWillChangeFrame:) name:UIKeyboardWillChangeFrameNotification object:nil];  
  3. }  
  4. - (void)closeNotification  
  5. {  
  6.     [[NSNotificationCenter defaultCenter] removeObserver:self name:UIKeyboardWillChangeFrameNotification object:nil];  
  7. }

  9. - (void)keyboardWillChangeFrame:(NSNotification *)notification {  
  10.       
  11.     NSDictionary *dict = [notification userInfo];  
  12.     // 键盘弹出和收回的时间  
  13.     CGFloat duration = [[dict objectForKey:UIKeyboardAnimationDurationUserInfoKey] floatValue];  
  14.     // 键盘初始时刻的frame  
  15.     CGRect beginKeyboardRect = [[dict objectForKey:UIKeyboardFrameBeginUserInfoKey] CGRectValue];  
  16.     // 键盘停止后的frame  
  17.     CGRect endKeyboardRect = [[dict objectForKey:UIKeyboardFrameEndUserInfoKey] CGRectValue];  
  18.     // 相减为键盘高度  
  19.     CGFloat yOffset = endKeyboardRect.origin.y - beginKeyboardRect.origin.y;  
  20.       
  21.     // 创建appDelegate单例对象  
  22.     AppDelegate *appDelegate = (AppDelegate *)[UIApplication sharedApplication].delegate;  
  23.     // 初始化一个数组,UIWindow的所有子视图  
  24.     NSArray *array = appDelegate.window.subviews;  
  25.     // 获取当前Controller的view视图  
  26.     UIView *view = appDelegate.window.subviews[array.count - 1];  
  27.     // textField相对于UIWindow的frame  
  28.     CGRect selfFrameFromUIWindow = [self convertRect:self.bounds toView:appDelegate.window];  
  29.     // textField底部距离屏幕底部的距离  
  30.     CGFloat bottomHeight = [UIScreen mainScreen].bounds.size.height - selfFrameFromUIWindow.origin.y - selfFrameFromUIWindow.size.height;  
  31.       
  32.       
  33.     // 初始化一个frame,大小为UIWindow的frame  
  34.     CGRect windowFrame = appDelegate.window.frame;  
  35.     // 把这个frame的y值增加或减少相应的高度(这里的40是textField底部和键盘顶部的距离)  
  36.     windowFrame.origin.y += yOffset + bottomHeight - 40;  
  37.     // 根据yOffset判断键盘是弹出还是收回  
  38.     if (yOffset < 0) {  
  39.         // 键盘弹出,改变当前Controller的view的frame  
  40.         [UIView animateWithDuration:duration animations:^{  
  41.             view.frame = windowFrame;  
  42.         }];  
  43.     } else {  
  44.         // 键盘收回,把view的frame恢复原状  
  45.         [UIView animateWithDuration:duration animations:^{  
  46.             view.frame = appDelegate.window.frame;  
  47.         }];  
  48.     }  
  49. }
