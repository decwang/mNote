读取图片中得二维码，我是这么一个思路，往方法里面传一个图片对象，这个方法就自动读取这个图片信息。

界面：

[objc] view plain copy
 
  1. #pragma mark - 读取图片中的二维码  
  2. /** 
  3.  *  读取图片中的二维码 
  4.  * 
  5.  *  @param image 图片 
  6.  * 
  7.  *  @return 图片中的二维码数据集合 CIQRCodeFeature对象 
  8.  */  
  9. + (NSArray *)readQRCodeFromImage:(UIImage *)image{  
  10.     // 创建一个CIImage对象  
  11.     CIImage *ciImage = [[CIImage alloc] initWithCGImage:image.CGImage options:nil];  
  12.     CIContext *context = [CIContext contextWithOptions:@{kCIContextUseSoftwareRenderer : @(YES)}]; // 软件渲染  
  13.     CIDetector *detector = [CIDetector detectorOfType:CIDetectorTypeQRCode context:context options:@{CIDetectorAccuracy : CIDetectorAccuracyHigh}];// 二维码识别  
  14.     // 注意这里的CIDetectorTypeQRCode  
  15.     NSArray *features = [detector featuresInImage:ciImage];  
  16.     NSLog(@"features = %@",features); // 识别后的结果集  
  17.     for (CIQRCodeFeature *feature in features) {  
  18.         NSLog(@"msg = %@",feature.messageString); // 打印二维码中的信息  
  19.     }  
  20.     return features;  
  21. }

截图方法：

[objc] view plain copy
 
  1. /** 
  2.  *  截图 
  3.  * 
  4.  *  @param view 需要截取的视图 
  5.  * 
  6.  *  @return 截图 图片 
  7.  */  
  8. + (UIImage *)screenShotFormView:(UIView *)view  
  9. {  
  10.     UIGraphicsBeginImageContextWithOptions(view.bounds.size, YES, [UIScreen mainScreen].scale);  
  11.     [view.layer renderInContext:UIGraphicsGetCurrentContext()];  
  12.     UIImage *img = UIGraphicsGetImageFromCurrentImageContext();  
  13.     UIGraphicsEndImageContext();  
  14.     return img;  
  15. }
