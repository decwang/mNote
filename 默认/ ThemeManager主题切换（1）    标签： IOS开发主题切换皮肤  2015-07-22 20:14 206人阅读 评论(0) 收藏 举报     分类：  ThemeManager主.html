  108. //4.根据传入主题名称来获取主题图片  

  109. - (UIImage *)getThemeImg:(NSString *)imgName {  

  110.       

  111. //    如果传入空值  

  112.     if (imgName.length == 0) {  

  113.         return nil;  

  114.     }  

  115. //    获取主题包得路径/....skins/cat/图片名称（获得图片路径）  

  116.     NSString *imagePath = [[self themePath] stringByAppendingPathComponent:imgName];  

  117. //    通过路径获得图片  

  118.     UIImage *img = [UIImage imageWithContentsOfFile:imagePath];  

  119.       

  120.     return img;  

  121. }

  123. //5.根据传入的字体颜色的key来获取主题字体  

  124. -(UIColor *)getThemeColor:(NSString *)themeKeyName  

  125. {  

  126.       

  127. //    如果传入空值  

  128.     if (themeKeyName.length == 0) {  

  129.         return nil;  

  130.     }  

  131.     //通过主题key取值 还是字典  

  132.     NSDictionary *rgbDic = [_fontColorDic objectForKey:themeKeyName];  

  133.     CGFloat r = [rgbDic[@"R"] floatValue];  

  134.     CGFloat g = [rgbDic[@"G"] floatValue];  

  135.     CGFloat b = [rgbDic[@"B"] floatValue];  

  136.     CGFloat alpha = [rgbDic[@"alpha"] floatValue];  

  137.       

  138.     if (rgbDic[@"alpha"] == nil) {  

  139.         alpha = 1;  

  140.     }  

  141.       

  142.     return [UIColor colorWithRed:r/255 green:g/255 blue:b/255 alpha:1];  

  143. }  

  144.   

  145. //获得主题包  

  146. - (NSString *)themePath  

  147. {  

  148.     //    取得主题路径  

  149.     //（1）取得程序包的根路径  

  150.     NSString *resourcePath = [[NSBundle mainBundle] resourcePath];  

  151.       

  152.     //（2）获取图片的路径 skins/cat  

  153.     NSString *imgPath = [_themeDic objectForKey:_themeName];  

  154.       

  155.     //(3)获取主题包得路径/....skins/cat  

  156.     return [resourcePath stringByAppendingPathComponent:imgPath];  

  157.       

  158. }  

  159.   

  160. //保存主题  

  161. -(void)saveTheme  

  162. {  

  163.     [[NSUserDefaults standardUserDefaults] setObject:_themeName forKey:kThemeName];  

  164.     [[NSUserDefaults standardUserDefaults] synchronize];  

  165. }
