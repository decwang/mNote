在QQ上我们经常会看到进入聊天页面时，输入框会跟着键盘的弹起而上下移动，本demo就是仿照此功能而设计

代码如下：

  1. #pragma mark- vc  
  2. - (void)viewDidLoad {  
  3.     [super viewDidLoad];  
  4.     self.title = @"与xxx聊天中";  
  5.       
  6.         //从IOS7开始，导航条和状态栏合为一体，而且呈半透明状，viewController.view的坐标0，0点在屏幕左上方，如果子视图位置在view的顶部，就会被导航条遮盖，所以，viewController会对自身的第一个子视图(scrollView)顶部添加64像素的contentInset，避免scrollView内容被查盖。  
  7.     self.automaticallyAdjustsScrollViewInsets = NO;  
  8.

  11.     _table.dataSource = self;  
  12.       
  13.         //设置分割线的样式。  
  14.     _table.separatorStyle = UITableViewCellSeparatorStyleNone;  
  15.       
  16.         //scrollView滑动时键盘消失；

  55. #pragma mark- notificationMethod  
  56. - (void)receiveKeyboardFrameChangeNotification:(NSNotification *)noti{  
  57.         //NSLog(@"%@",noti.userInfo);  
  58.       
  59.         //NSValue值类,基本类型和结构体不能直接放入数组和字典中，如果需要把结构体放入数组或字典，那么必须把结构体转换为NSValue，然后把NSValue放入字典或数组。  
  60.     NSValue *value = [noti.userInfo objectForKey:UIKeyboardFrameEndUserInfoKey];  
  61.       
  62.         //CGRect rect = [value CGRectValue];  
  63.     CGRect rect;  
  64.     [value getValue:&rect];  
  65.       
  66.       
  67.     [UIView beginAnimations:nil context:nil];  
  68.     [UIView setAnimationDuration:[[noti.userInfo objectForKey:UIKeyboardAnimationDurationUserInfoKey] floatValue]];  
  69.         //7是键盘动画专用速率  
  70.     [UIView setAnimationCurve:[[noti.userInfo objectForKey:UIKeyboardAnimationCurveUserInfoKey] integerValue]];  
  71.     _inputView.frame = CGRectMake(0, rect.origin.y-40, 375, 40);  
  72.     _table.frame = CGRectMake(0, 0, 375, _inputView.frame.origin.y);  
  73.     if (_array.count>1) {  
  74.         [_table scrollToRowAtIndexPath:[NSIndexPath indexPathForRow:_array.count-1 inSection:0] atScrollPosition:UITableViewScrollPositionNone animated:NO];  
  75.     }  
  76.     [UIView commitAnimations];  
  77. }

  87. - (BOOL)textFieldShouldReturn:(UITextField *)textField{

  96. }

  102.   
  103. - (UITableViewCell *)tableView:(UITableView *)tableView cellForRowAtIndexPath:(NSIndexPath *)indexPath{  
  104.       
  105.     NSString *text = [_array objectAtIndex:indexPath.row];  
  106.         //boundingRectWithSize计算出文本显示的边框大小。第一个参数是限定的宽度和高度。第二个参数是计算选项，第三个参数是计算的文本的具体属性。  
  107.     NSMutableParagraphStyle *style = [[NSMutableParagraphStyle alloc] init];  
  108.     style.lineBreakMode = NSLineBreakByCharWrapping;  
  109.     CGRect rect = [text boundingRectWithSize:CGSizeMake(220, 0) options:NSStringDrawingUsesLineFragmentOrigin|NSStringDrawingUsesFontLeading attributes:@{NSFontAttributeName:[UIFont systemFontOfSize:17],NSParagraphStyleAttributeName:style} context:nil];  
  110.     [style release];  
  111.       
  112.     if (indexPath.row%2 == 0) {  
  113.         SelfCell *cell = [tableView dequeueReusableCellWithIdentifier:@"self"];  
  114.         cell.messageLabel.frame = CGRectMake(30, 10, rect.size.width, rect.size.height);  
  115.         cell.bubbleImageView.frame = CGRectMake(20, 5, rect.size.width+20, rect.size.height+10);  
  116.         cell.messageLabel.text = text;  
  117.         return cell;  
  118.     }else{  
  119.         OtherCell *cell = [tableView dequeueReusableCellWithIdentifier:@"other"];  
  120.         cell.messageLabel.frame = CGRectMake(375-rect.size.width-30, 10, rect.size.width, rect.size.height);  
  121.         cell.bubbleImageView.frame = CGRectMake(375-rect.size.width-40, 5, rect.size.width+20, rect.size.height+10);  
  122.         cell.messageLabel.text = text;  
  123.         return cell;  
  124.     }  
  125. }  
  126.   
  127. - (CGFloat)tableView:(UITableView *)tableView heightForRowAtIndexPath:(NSIndexPath *)indexPath{  
  128.     NSString *text = [_array objectAtIndex:indexPath.row];  
  129.     CGRect rect = [text boundingRectWithSize:CGSizeMake(220, 0) options:NSStringDrawingUsesLineFragmentOrigin|NSStringDrawingUsesFontLeading attributes:@{NSFontAttributeName:[UIFont systemFontOfSize:17]} context:nil];  
  130.     return rect.size.height+20;  
  131. }
