iOS UITextField随着键盘的高度变化动态调整位置

2013-06-29 17:22 1632人阅读 评论(0) 收藏 举报

在使用中，可以直接把下面的类复制到项目中，在创建文本框的类中添加如下代码：

 self.field=[[UITextField_UIKeyBoard alloc]init];
    CGRect fieldRect=CGRectMake(0, 420, 250, 40);
    if(IS_IPHONE_5)
    {
        fieldRect=CGRectMake(0, 526, 250, 40);
    }
    self.field.rect=fieldRect;
   [self.field textFieldFrame:self.field.rect viewController:self];

 

提到的类为：

  13.            (UIViewController *)viewController  
  14. {  
  15.     self.myTextField=[[UITextField alloc] init];//初始化UITextField  
  16.     self.myTextField.frame =self.rect;  
  17.     self.myTextField.delegate=self;//设置代理  
  18.     self.myTextField.borderStyle = UITextBorderStyleRoundedRect;  
  19.     self.myTextField.contentVerticalAlignment = UIControlContentVerticalAlignmentCenter;//垂直居中  
  20.     self.myTextField.placeholder = @"Please entry your content!";//内容为空时默认文字  
  21.     self.myTextField.returnKeyType = UIReturnKeyDone;//设置放回按钮的样式  
  22.     [viewController.view addSubview:self.myTextField];  
  23.     //注册键盘出现与隐藏时候的通知  
  24.     [[NSNotificationCenter defaultCenter] addObserver:self  
  25.                                              selector:@selector(keyboadWillShow:)  
  26.                                                  name:UIKeyboardWillShowNotification  
  27.                                                object:nil];  
  28.     [[NSNotificationCenter defaultCenter] addObserver:self  
  29.                                              selector:@selector(keyboardWillHide:)  
  30.                                                  name:UIKeyboardWillHideNotification  
  31.                                                object:nil];  
  32.     //添加手势，点击屏幕其他区域关闭键盘的操作  
  33.     UITapGestureRecognizer *gesture = [[UITapGestureRecognizer alloc] initWithTarget:self action:@selector(hideKeyboard)];  
  34.     gesture.numberOfTapsRequired = 1;//手势敲击的次数  
  35.     [viewController.view addGestureRecognizer:gesture];  
  36. }  
  37.   
  38. //键盘出现时候调用的事件  
  39. -(void) keyboadWillShow:(NSNotification *)note{  
  40.     NSDictionary *info = [note userInfo];  
  41.     CGSize keyboardSize = [[info objectForKey:UIKeyboardFrameEndUserInfoKey] CGRectValue].size;//键盘的frame  
  42.     CGFloat offY = ([UIScreen mainScreen].bounds.size.height-keyboardSize.height)-self.myTextField.frame.size.height;//屏幕总高度-键盘高度-UITextField高度  
  43.     [UIView beginAnimations:nil context:NULL];//此处添加动画，使之变化平滑一点  
  44.     [UIView setAnimationDuration:0.3];//设置动画时间 秒为单位  
  45.     self.myTextField.frame = CGRectMake(self.rect.origin.x, offY, self.rect.size.width, self.rect.size.height);//UITextField位置的y坐标移动到offY  
  46.     [UIView commitAnimations];//开始动画效果  
  47.       
  48. }  
  49. //键盘消失时候调用的事件  
  50. -(void)keyboardWillHide:(NSNotification *)note{  
  51.     [UIView beginAnimations:nil context:NULL];//此处添加动画，使之变化平滑一点  
  52.     [UIView setAnimationDuration:0.3];  
  53.     self.myTextField.frame = self.rect;//UITextField位置复原  
  54.       
  55.     [UIView commitAnimations];  
  56. }  
  57. //隐藏键盘方法  
  58. -(void)hideKeyboard{  
  59.     [self.myTextField resignFirstResponder];  
  60. }  
  61. //点击return按钮所做的动作：  
  62. - (BOOL)textFieldShouldReturn:(UITextField *)textField  
  63. {  
  64.     [textField resignFirstResponder];//取消第一响应  
  65.     return YES;  
  66. }
